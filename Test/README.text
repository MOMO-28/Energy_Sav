            const select = document.getElementById('selectItem');

            function createTable(headers, rows) {
                const table = document.createElement('table');
                const thead = document.createElement('thead');
                const trHead = document.createElement('tr');
                headers.forEach(headerText => {
                const th = document.createElement('th');
                th.textContent = headerText;
                trHead.appendChild(th);
                });
                thead.appendChild(trHead);
                table.appendChild(thead);

                const tbody = document.createElement('tbody');
                rows.forEach(row => {
                const tr = document.createElement('tr');
                row.forEach(cellText => {
                    const td = document.createElement('td');
                    td.innerHTML = cellText;
                    tr.appendChild(td);
                });
                tbody.appendChild(tr);
                });
                table.appendChild(tbody);
                return table;
            }

            select.addEventListener('change', () => {
                const val = select.value;
                if (!val || !contentData[val]) {
                contentDiv.innerHTML = '';
                return;
                }

                const data = contentData[val];
                contentDiv.innerHTML = ''; // クリア

                // 画像表示
                if (data.img) {
                const img = document.createElement('img');
                img.src = data.img;
                img.alt = data.label || '';
                contentDiv.appendChild(img);
                }

                // テキスト表示（HTML形式で入れているのでinnerHTML）
                if (data.text) {
                const p = document.createElement('p');
                p.innerHTML = data.text;
                contentDiv.appendChild(p);
                }

                // テーブル表示
                if (data.table) {
                const table = createTable(data.table.headers, data.table.rows);
                contentDiv.appendChild(table);
                }
            });
